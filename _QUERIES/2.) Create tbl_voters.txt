-- Drop old table
DROP TABLE tbl_voters;

-- Recreate with correct column order
CREATE TABLE tbl_voters (
    user_id        NUMBER PRIMARY KEY,
    student_id     VARCHAR2(50) UNIQUE NOT NULL,
    voters_id      VARCHAR2(10) UNIQUE NOT NULL,
    gender         VARCHAR2(10),
    birthday       VARCHAR2(20),
    age            NUMBER(3),
    course         VARCHAR2(20),
    created_at     VARCHAR2(30),
    is_voted       NUMBER(1)     DEFAULT 0,
    otp_code       VARCHAR2(255),
    otp_expires_at VARCHAR2(30)
);

-- Recreate trigger
CREATE OR REPLACE TRIGGER voters_user_id_trg
BEFORE INSERT ON tbl_voters
FOR EACH ROW
BEGIN
    IF :NEW.user_id IS NULL THEN
        SELECT voters_user_id_seq.NEXTVAL INTO :NEW.user_id FROM dual;
    END IF;
END;

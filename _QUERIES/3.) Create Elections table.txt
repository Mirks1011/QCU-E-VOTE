-- Sequence for election_id
CREATE SEQUENCE elections_id_seq
    START WITH 1
    INCREMENT BY 1
    NOCACHE
    NOCYCLE;

-- Create elections table
CREATE TABLE tbl_elections (
    election_id          VARCHAR2(10)  PRIMARY KEY,
    election_title       VARCHAR2(100) NOT NULL,
    election_description VARCHAR2(500),
    start_datetime       VARCHAR2(30),
    end_datetime         VARCHAR2(30),
    status               VARCHAR2(20)  DEFAULT 'scheduled',
    created_at           VARCHAR2(30),
    created_by           VARCHAR2(50)
);

-- Trigger for election_id auto-generate in format eXX-XXXX
CREATE OR REPLACE TRIGGER elections_id_trg
BEFORE INSERT ON tbl_elections
FOR EACH ROW
BEGIN
    IF :NEW.election_id IS NULL THEN
        DECLARE
            seq_val NUMBER;
        BEGIN
            SELECT elections_id_seq.NEXTVAL INTO seq_val FROM dual;
            :NEW.election_id := 'e'
                || LPAD(FLOOR((seq_val - 1) / 10000) + 1, 2, '0')
                || '-'
                || LPAD(MOD(seq_val - 1, 10000) + 1, 4, '0');
        END;
    END IF;
END;
